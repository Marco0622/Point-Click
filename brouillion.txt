onst head = document.querySelectorAll(".map img");
  const scoreBoard = document.querySelector('.score')
  let intKill = 0;

  head.forEach(m => {
    m.addEventListener('click', () => {
      
      // Récupère la position actuelle de l'élément dans la fenêtre
      const rect2 = m.getBoundingClientRect();
      
      // Fige l'élément à sa position actuelle
      m.style.left = rect2.left + 'px';
      m.style.top = rect2.top + 'px';
      m.style.position = "fixed";

      // Ajoute une classe qui lance l'animation de "kill"
      m.classList.add("kill");

      // Une fois l'animation terminée
      m.addEventListener("animationend", () => {
        
        respawn(m);
        // Retire la classe "kill"
        m.classList.remove("kill");

        // Remet l'élément en position absolue pour qu'il reprenne son comportement normal
        m.style.position = "absolute";

        // Supprime les positions left/top fixées précédemment
        m.style.removeProperty('left');
        m.style.removeProperty('top');

        function respawn(m){
          const zone = document.getElementById("map");
          const rect = zone.getBoundingClientRect();
          const size = m.getBoundingClientRect().width;

          m.targetX = Math.random() *( rect.width - size);
          m.targetY = Math.random() * (rect.height - size);

          m.x = Math.random() * rect.width;
          m.y = Math.random() * rect.height;
        }

      }, { once: true }); // L'événement ne sera écouté qu'une seule fois

      // Incrémente le score
      intKill++;

      // Met à jour l'affichage du score
      scoreBoard.textContent = intKill;
    });
});
});
